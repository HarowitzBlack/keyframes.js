Press F12 and have fun!

The keyframes library can be accessed via K global variable.
The shape ids are #shape-1, #shape-2, #shape-3 and so on.

...or just play around with examples below...

Hide/show

var hide = K.tween({ transform: "scale(1) rotate(0deg)", opacity: 1 }, { transform: "scale(0.1) rotate(90deg)", opacity: 0 }); var hideThenShow = K.toAndFrom(hide); K.stream(2000, K.easings.easeInOutExpo(hideThenShow), K.intoDom(document.getElementById('shape-1')));

Hide/show

var roll = K.tween({ transformOrigin: "100% 100%", transform: "rotateZ(0deg)" }, { transformOrigin: "100% 100%", transform: "rotateZ(90deg)" }); var roll4times = K.repeat(4, roll); K.stream(1000, roll4times, K.intoDom(document.getElementById("shape-2")));
